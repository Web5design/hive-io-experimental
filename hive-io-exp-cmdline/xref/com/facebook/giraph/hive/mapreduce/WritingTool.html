49  public class WritingTool extends Configured implements Tool {
50    private static final Logger LOG = Logger.getLogger(WritingTool.class);
51  
52    @Override
53    public int run(String[] args) throws Exception {
54      Configuration conf = getConf();
55      handleCommandLine(args, conf);
56      HadoopUtils.setMapAttempts(conf, 1);
57      adjustConfigurationForHive(conf);
58      HiveTools.setupJob(conf);
59  
60      Job job = new Job(conf, "hive-io-writing");
61      if (job.getJar() == null) {
62        job.setJarByClass(getClass());
63      }
64      job.setMapperClass(SampleMapper.class);
65      job.setInputFormatClass(SampleInputFormat.class);
66      job.setMapOutputKeyClass(NullWritable.class);
67      job.setMapOutputValueClass(HiveWritableRecord.class);
68      job.setOutputFormatClass(SampleOutputFormat.class);
69  
70      job.setNumReduceTasks(0);
71  
72      job.submit();
73      return job.waitForCompletion(true) ? 0 : 1;
74    }

76    /**
77     * set hive configuration
78     */
79    private void adjustConfigurationForHive(Configuration conf) {
80      // when output partitions are used, workers register them to the
81      // metastore at cleanup stage, and on HiveConf's initialization, it
82      // looks for hive-site.xml.
83      addToStringCollection(
84        conf, "tmpfiles",
85        conf.getClassLoader().getResource("hive-site.xml").toString()
86      );
87  
88      // Or, more effectively, we can provide all the jars client needed to
89      // the workers as well
90      String[] hadoopJars = System.getenv("HADOOP_CLASSPATH").split(
91        File.pathSeparator
92      );
93      List<String> hadoopJarURLs = Lists.newArrayList();
94      for (String jarPath : hadoopJars) {
95        File file = new File(jarPath);
96        if (file.exists() && file.isFile()) {
97          String jarURL = file.toURI().toString();
98          hadoopJarURLs.add(jarURL);
99        }
100     }
101     addToStringCollection(conf, "tmpjars", hadoopJarURLs);
102   }

111   private static void addToStringCollection(

113     String... values

115     addToStringCollection(conf, name, Arrays.asList(values));
116   }

118   /**
119    * add string to collection
120    *
121    * @param conf   Configuration
122    * @param name   to add
123    * @param values values for collection
124    */
125   private static void addToStringCollection(
126     Configuration conf, String name,
127     Collection<? extends String> values
128   ) {
129     Collection<String> tmpfiles = conf.getStringCollection(name);
130     tmpfiles.addAll(values);
131     conf.setStrings(name, tmpfiles.toArray(new String[tmpfiles.size()]));
132   }

134   /**
135    * process arguments
136    *
137    * @param args to process
138    * @return CommandLine instance
139    * @throws org.apache.commons.cli.ParseException
140    *                              error parsing arguments
141    * @throws InterruptedException interrupted
142    */
143   private void handleCommandLine(String[] args, Configuration conf) throws ParseException {
144     Options options = new Options();
145     options.addOption(
146       "D", "hiveconf", true,
147       "property=value for Hive/Hadoop configuration"
148     );
149     CommandLineParser parser = new GnuParser();
150     CommandLine cmdln = parser.parse(options, args);
151     // pick up -hiveconf arguments
152     processHiveConfOptions(cmdln, conf);
153   }
154 
155   /**
156    * Process -hiveconf/-D options from command line
157    *
158    * @param cmdln Command line options
159    */
160   private void processHiveConfOptions(CommandLine cmdln, Configuration conf) {
161     for (String hiveconf : cmdln.getOptionValues("hiveconf")) {
162       String[] keyval = hiveconf.split("=", 2);
163       if (keyval.length == 2) {
164         String name = keyval[0];
165         String value = keyval[1];
166         if (name.equals("tmpjars") || name.equals("tmpfiles")) {
167           addToStringCollection(conf, name, value);
168         } else {
169           conf.set(name, value);
170         }
171       }
172     }
173   }
