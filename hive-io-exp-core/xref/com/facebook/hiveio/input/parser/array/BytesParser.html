30  public class BytesParser implements RecordParser {
31    private final int[] columnIndexes;
32    private final ArrayRecord record;
33  
34    public BytesParser(String[] partitionValues, int numColumns, ArrayParserData parserData) {
35      columnIndexes = parserData.columnIndexes;
36      record = new ArrayRecord(numColumns, partitionValues, parserData.hiveTypes);
37    }
38  
39    @Override
40    public HiveReadableRecord createRecord() {
41      return record;
42    }
43  
44    @Override
45    public HiveReadableRecord parse(Writable value, HiveReadableRecord record) throws IOException {
46      final BytesRefArrayWritable braw = (BytesRefArrayWritable) value;
47      final ArrayRecord arrayRecord = (ArrayRecord) record;
48  
49      arrayRecord.reset();
50  
51      for (int i = 0; i < columnIndexes.length; i++) {
52        final int column = columnIndexes[i];
53        final BytesRefWritable fieldData = braw.unCheckedGet(column);
54  
55        final byte[] bytes = fieldData.getData();
56        final int start = fieldData.getStart();
57        final int length = fieldData.getLength();
58  
59        if (length == "\\N".length() && bytes[start] == '\\' &&
60            bytes[start + 1] == 'N') {
61          arrayRecord.setNull(column, true);
62        } else {
63          parsePrimitiveColumn(column, bytes, start, length);
64        }
65      }
66  
67      return arrayRecord;
68    }
69  
70    private void parsePrimitiveColumn(int column, byte[] bytes, int start, int length) {
71      switch (record.getNativeType(column)) {
72        case BOOLEAN:
73          Boolean bool = parseBoolean(bytes, start, length);
74          if (bool == null) {
75            record.setNull(column, true);
76          } else {
77            record.setBoolean(column, bool);
78          }
79          break;
80        case LONG:
81          if (length == 0) {
82            record.setNull(column, true);
83          } else {
84            record.setLong(column, parseLong(bytes, start, length));
85          }
86          break;
87        case DOUBLE:
88          if (length == 0) {
89            record.setNull(column, true);
90          } else {
91            record.setDouble(column, parseDouble(bytes, start, length));
92          }
93          break;
94        case STRING:
95          record.setString(column, parseString(bytes, start, length));
96          break;
97        default:
98          break;
99      }
100   }

102   public static String parseString(byte[] bytes, int start, int length) {
103     return new String(bytes, start, length, Charsets.UTF_8);
104   }

106   public static Boolean parseBoolean(byte[] bytes, int start, int length) {
107     if ((length == 4) &&
108         (toUpperCase(bytes[start + 0]) == 'T') &&
109         (toUpperCase(bytes[start + 1]) == 'R') &&
110         (toUpperCase(bytes[start + 2]) == 'U') &&
111         (toUpperCase(bytes[start + 3]) == 'E')) {
112       return true;
113     }
114     if ((length == 5) &&
115         (toUpperCase(bytes[start + 0]) == 'F') &&
116         (toUpperCase(bytes[start + 1]) == 'A') &&
117         (toUpperCase(bytes[start + 2]) == 'L') &&
118         (toUpperCase(bytes[start + 3]) == 'S') &&
119         (toUpperCase(bytes[start + 4]) == 'E')) {
120       return false;
121     }
122     return null;
123   }

125   private static byte toUpperCase(byte b) {
126     return isLowerCase(b) ? ((byte) (b - 32)) : b;
127   }
128 
129   private static boolean isLowerCase(byte b) {
130     return (b >= 'a') && (b <= 'z');
131   }
132 
133   public static long parseLong(byte[] bytes, int start, int length) {
134     int limit = start + length;
135 
136     int sign = bytes[start] == '-' ? -1 : 1;
137 
138     if (sign == -1 || bytes[start] == '+') {
139       start++;
140     }
141 
142     long value = bytes[start++] - '0';
143     while (start < limit) {
144       value = value * 10 + (bytes[start] - '0');
145       start++;
146     }
147 
148     return value * sign;
149   }
150 
151   public static double parseDouble(byte[] bytes, int start, int length) {
152     char[] chars = new char[length];
153     for (int pos = 0; pos < length; pos++) {
154       chars[pos] = (char) bytes[start + pos];
155     }
156     String string = new String(chars);
157     return Double.parseDouble(string);
158   }
