39  /**
40   * Single record from a Hive table. Used for both reading and writing.
41   */
42  public class DefaultRecord implements HiveRecord {
43    /** Logger */
44    private static final Logger LOG = LoggerFactory.getLogger(DefaultRecord.class);
45  
46    /** Partition data */
47    private final String[] partitionValues;
48    /** Raw data for row */
49    private final Object[] rowData;
50  
51    /**
52     * Constructor
53     *
54     * @param numColumns number of columns
55     * @param partitionValues partition data
56     */
57    public DefaultRecord(int numColumns, String[] partitionValues) {
58      this.partitionValues = partitionValues;
59      this.rowData = new Object[numColumns];
60    }
61  
62    @Override
63    public Object get(int index) {
64      if (index < rowData.length) {
65        return rowData[index];
66      } else {
67        return partitionValues[index - rowData.length];
68      }
69    }
70  
71    @Override
72    public boolean getBoolean(int index) {
73      Boolean v = (Boolean) get(index);
74      return v == null ? false : v;
75    }
76  
77    @Override
78    public long getLong(int index) {
79      Long v = (Long) get(index);
80      return v == null ? Long.MIN_VALUE : v;
81    }
82  
83    @Override
84    public double getDouble(int index) {
85      Double v = (Double) get(index);
86      return v == null ? Double.NaN : v;
87    }
88  
89    @Override
90    public String getString(int index) {
91      return (String) get(index);
92    }
93  
94    @Override
95    public boolean isNull(int index) {
96      return get(index) == null;
97    }
98  
99    @Override
100   public List<Object> getAllColumns() {
101     return Arrays.asList(rowData);
102   }
103 
104   @Override
105   public void set(int index, Object value) {
106     rowData[index] = value;
107   }
108 
109   public int getNumColumns() {
110     return rowData.length;
111   }
112 
113   /**
114    * Parse a row
115    *
116    * @param value Row from Hive
117    * @param deserializer Deserializer
118    * @throws IOException I/O errors
119    */
120   public void parse(Writable value, Deserializer deserializer)
121     throws IOException {
122     Object data;
123     ObjectInspector dataInspector;
124     try {
125       data = deserializer.deserialize(value);
126       dataInspector = deserializer.getObjectInspector();
127     } catch (SerDeException e) {
128       throw new IOException(e);
129     }
130 
131     Preconditions.checkArgument(dataInspector.getCategory() == Category.STRUCT);
132     StructObjectInspector structInspector =
133         (StructObjectInspector) dataInspector;
134 
135     Object parsedData = ObjectInspectorUtils.copyToStandardJavaObject(data,
136         structInspector);
137     List<Object> parsedList = (List<Object>) parsedData;
138     for (int i = 0; i < parsedList.size(); ++i) {
139       set(i, parsedList.get(i));
140     }
141   }
142 
143   @Override
144   public String toString() {
145     return Objects.toStringHelper(this)
146         .add("numColumns", getNumColumns())
147         .add("rowData", rowDataToString())
148         .add("partitionData", partitionValues)
149         .toString();
150   }
151 
152   /**
153    * String representation of row data for debugging
154    * @return String dump of row data
155    */
156   private String rowDataToString() {
157     StringBuilder sb = new StringBuilder();
158     sb.append('[');
159     for (int i = 0; i < rowData.length; ++i) {
160       if (i > 0) {
161         sb.append(',');
162       }
163       if (rowData[i] == null) {
164         sb.append("(null)");
165       } else {
166         sb.append(rowData[i].getClass().getSimpleName());
167         sb.append(":");
168         sb.append(rowData[i].toString());
169       }
170     }
171     sb.append(']');
172     return sb.toString();
173   }
