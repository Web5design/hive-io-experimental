46  /**
47   * Holds information for Hive output
48   */
49  class OutputInfo implements Writable {
50    /** Logger */
51    public static final Logger LOG = Logger.getLogger(OutputInfo.class);
52  
53    /** Parameters for Hive table */
54    private final Map<String, String> tableParams;
55  
56    /** Class for writing */
57    private Class<? extends OutputFormat> outputFormatClass;
58  
59    /** Partition column information */
60    private final List<FieldSchema> partitionInfo;
61    /** Regular column information */
62    private final List<FieldSchema> columnInfo;
63  
64    /** Class used for serialization */
65    private Class<? extends Serializer> serializerClass;
66    /** Parameters for serializer */
67    private final Map<String, String> serializerParams;
68  
69    /** Path to table root in Hadoop */
70    private String tableRoot;
71    /**
72     * Path to specific partition we're writing. If not partitioned, this will be
73     * the same as tablePath
74     */
75    private String partitionPath;
76    /**
77     * Path to where we're writing to. If partitioned this is the same as
78     * partitionPath, otherwise this is a temporary path.
79     */
80    private String finalOutputPath;
81  
82    /**
83     * Default constructor
84     */
85    public OutputInfo() {
86      this.tableParams = Maps.newHashMap();
87      this.partitionInfo = Lists.newArrayList();
88      this.columnInfo = Lists.newArrayList();
89      this.serializerClass = null;
90      this.serializerParams = Maps.newHashMap();
91    }
92  
93    /**
94     * Construct from Hive table
95     * @param table Hive table to grab information from
96     */
97    public OutputInfo(Table table) {
98      partitionInfo = table.getPartitionKeys();
99  
100     StorageDescriptor storageDescriptor = table.getSd();
101     tableParams = table.getParameters();
102     outputFormatClass =
103         Classes.classForName(storageDescriptor.getOutputFormat());
104     columnInfo = storageDescriptor.getCols();
105     tableRoot = storageDescriptor.getLocation();
106 
107     SerDeInfo serDeInfo = storageDescriptor.getSerdeInfo();
108     serializerClass = SerDes.getSerDeClass(serDeInfo);
109     serializerParams = serDeInfo.getParameters();
110   }

164   /**
165    * Create Serializer using Configuration passed in
166    *
167    * @param conf Configuration to use
168    * @return A new, configured, Serializer
169    */
170   public Serializer createSerializer(Configuration conf) {
171     Serializer serializer = ReflectionUtils.newInstance(serializerClass, conf);
172     SerDes.initSerializer(serializer, conf, columnInfo, serializerParams);
173     return serializer;
174   }

176   @Override
177   public void write(DataOutput out) throws IOException {
178     Writables.writeClassName(out, Preconditions.checkNotNull(outputFormatClass));
179     Writables.writeFieldSchemas(out, partitionInfo);
180     Writables.writeStrStrMap(out, tableParams);
181     Writables.writeFieldSchemas(out, columnInfo);
182     WritableUtils.writeString(out, Preconditions.checkNotNull(tableRoot));
183     WritableUtils.writeString(out, Preconditions.checkNotNull(partitionPath));
184     WritableUtils.writeString(out, Preconditions.checkNotNull(finalOutputPath));
185     Writables.writeClassName(out, Preconditions.checkNotNull(serializerClass));
186     Writables.writeStrStrMap(out, serializerParams);
187   }
188 
189   @Override
190   public void readFields(DataInput in) throws IOException {

192     Writables.readFieldSchemas(in, partitionInfo);
193     Writables.readStrStrMap(in, tableParams);
194     Writables.readFieldSchemas(in, columnInfo);
195     tableRoot = WritableUtils.readString(in);
196     partitionPath = WritableUtils.readString(in);
197     finalOutputPath = WritableUtils.readString(in);
198     serializerClass = Writables.readClass(in);
199     Writables.readStrStrMap(in, serializerParams);
200   }
