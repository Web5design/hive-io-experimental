150   public List<InputSplit> getSplits(HiveConf conf, ThriftHiveMetastore.Iface client)
151       throws IOException {
152     HiveInputDescription inputDesc = readProfileInputDesc(conf);
153 
154     Table table;
155     try {
156       table = client.get_table(inputDesc.getDbName(), inputDesc.getTableName());
157     } catch (Exception e) {
158       throw new IOException(e);
159     }
160 
161     final HiveTableSchema tableSchema = HiveTableSchemaImpl.fromTable(table);
162     HiveTableSchemas.put(conf, myProfileId, tableSchema);
163 
164     List<InputPartition> partitions = computePartitions(inputDesc, client, table);
165 
166     List<InputSplit> splits = computeSplits(conf, inputDesc, tableSchema, partitions);
167 
168     return splits;
169   }

171   private List<InputSplit> computeSplits(HiveConf conf, HiveInputDescription inputDesc,
172     HiveTableSchema tableSchema, List<InputPartition> partitions) throws IOException
173   {
174     int partitionNum = 0;
175     List<InputSplit> splits = Lists.newArrayList();
176 
177     int[] columnIds = computeColumnIds(inputDesc.getColumns(), tableSchema);
178 
179     for (InputPartition inputPartition : partitions) {
180       org.apache.hadoop.mapred.InputFormat baseInputFormat = inputPartition.makeInputFormat(conf);
181       HadoopUtils.setInputDir(conf, inputPartition.getLocation());
182 
183       org.apache.hadoop.mapred.InputSplit[] baseSplits =
184           baseInputFormat.getSplits(new JobConf(conf), inputDesc.getNumSplits());
185       LOG.info("Requested " + inputDesc.getNumSplits() + " splits from partition (" +
186           partitionNum + " out of " + Iterables.size(partitions) +
187           ") values: " + inputPartition.getInputSplitData().getPartitionValues() +
188           ", got " + baseSplits.length + " splits from inputFormat " +
189           baseInputFormat.getClass().getCanonicalName());
190 
191       for (org.apache.hadoop.mapred.InputSplit baseSplit : baseSplits)  {
192         InputSplit split = new HInputSplit(baseInputFormat, baseSplit,
193             tableSchema, columnIds, inputPartition.getInputSplitData(), conf);
194         splits.add(split);
195       }
196 
197       partitionNum++;
198     }
199     return splits;
200   }

202   private int[] computeColumnIds(List<String> columnNames, HiveTableSchema tableSchema) {
203     List<Integer> ints;
204     if (columnNames.isEmpty()) {
205       Range<Integer> range = Ranges.closedOpen(0, tableSchema.numColumns());
206       ints = range.asSet(DiscreteDomains.integers()).asList();
207     } else {
208       ints = transform(columnNames, HiveTableSchemas
209           .schemaLookupFunc(tableSchema));
210     }
211     int[] result = new int[ints.size()];
212     for (int i = 0; i < ints.size(); ++i) {
213       result[i] = ints.get(i);
214     }
215     return result;
216   }

218   private List<InputPartition> computePartitions(HiveInputDescription inputDesc,
219     ThriftHiveMetastore.Iface client, Table table) throws IOException
220   {
221     List<InputPartition> partitions = Lists.newArrayList();
222 
223     if (table.getPartitionKeysSize() == 0) {
224       // table without partitions
225       partitions.add(InputPartition.newFromHiveTable(table));
226     } else {
227       // table with partitions, find matches to user filter.
228       List<Partition> hivePartitions;
229       try {
230         hivePartitions = client.get_partitions_by_filter(inputDesc.getDbName(),
231             inputDesc.getTableName(), inputDesc.getPartitionFilter(), (short) -1);
232       } catch (Exception e) {
233         throw new IOException(e);
234       }
235       for (Partition hivePartition : hivePartitions) {
236         partitions.add(InputPartition.newFromHivePartition(hivePartition));
237       }
238     }
239     return partitions;
240   }

242   @Override
243   public RecordReaderImpl
244   createRecordReader(InputSplit inputSplit, TaskAttemptContext context)
245     throws IOException, InterruptedException
246   {
247     Configuration conf = context.getConfiguration();
248     JobConf jobConf = new JobConf(conf);
249 
250     HInputSplit split = (HInputSplit) inputSplit;
251     split.setConf(jobConf);
252 
253     int[] columnIds = split.getColumnIds();
254     HiveUtils.setReadColumnIds(jobConf, columnIds);
255 
256     // CHECKSTYLE: stop LineLength
257     org.apache.hadoop.mapred.RecordReader<WritableComparable, Writable> baseRecordReader =
258         split.getBaseRecordReader(jobConf, context);
259     // CHECKSTYLE: resume LineLength
260 
261     RecordParser<Writable> recordParser = getParser(baseRecordReader.createValue(),
262         split.getTableSchema().getTableName(), split, columnIds);
263 
264     RecordReaderImpl reader = new RecordReaderImpl(baseRecordReader, recordParser);
265     reader.setObserver(observer);
266 
267     return reader;
268   }

270   private RecordParser<Writable> getParser(Writable exampleValue,
271     HiveTableName tableName, HInputSplit split, int[] columnIds)
272   {
273     Deserializer deserializer = split.getDeserializer();
274     String[] partitionValues = split.getPartitionValues();
275     int numColumns = split.getTableSchema().numColumns();
276     return Parsers.bestParser(deserializer, numColumns, columnIds, tableName,
277         partitionValues, exampleValue);
278   }
