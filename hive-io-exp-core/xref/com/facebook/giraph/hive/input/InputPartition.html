41  /**
42   * Internal partition information for reading
43   */
44  class InputPartition implements Writable {
45    /** Logger */
46    private static final Logger LOG = Logger.getLogger(InputPartition.class);
47  
48    /** Data for this partition */
49    private final InputSplitData inputSplitData;
50    /** Class for InputFormat to read this partition */
51    private Class<? extends InputFormat> inputFormatClass;
52    /** Location in Hadoop */
53    private String location;
54  
55    /** Constructor for reflection */
56    public InputPartition() {
57      inputSplitData = new InputSplitData();
58    }
59  
60    /**
61     * Constructor with data
62     * @param inputSplitData split data
63     * @param inputFormatClass class for input format
64     * @param location where data is on hdfs
65     */
66    public InputPartition(InputSplitData inputSplitData,
67                          Class<? extends InputFormat> inputFormatClass,
68                          String location) {
69      this.inputSplitData = inputSplitData;
70      this.inputFormatClass = inputFormatClass;
71      this.location = location;
72    }
73  
74    /**
75     * Create for un-partitioned tables
76     * @param table Hive table
77     * @return InputPartition
78     */
79    public static InputPartition newFromHiveTable(Table table) {
80      return makePartition(table.getSd(), Collections.<String>emptyList());
81    }
82  
83    /**
84     * Create for partition of a table
85     * @param partition Hive partition
86     * @return InputPartition
87     */
88    public static InputPartition newFromHivePartition(Partition partition) {
89      return makePartition(partition.getSd(), partition.getValues());
90    }
91  
92    /**
93     * Helper to construct class
94     * @param storageDescriptor {@link StorageDescriptor}
95     * @param partitionValues list of partition values
96     * @return InputPartition
97     */
98    private static InputPartition makePartition(StorageDescriptor storageDescriptor,
99                                                List<String> partitionValues) {
100     InputSplitData splitData = new InputSplitData(storageDescriptor, partitionValues);
101     Class<? extends InputFormat> inputFormatClass = Classes
102         .classForName(storageDescriptor.getInputFormat());
103     String location = storageDescriptor.getLocation();
104     return new InputPartition(splitData, inputFormatClass, location);
105   }
106 
107   /**
108    * Create InputFormat for this partition
109    *
110    * @param conf Configuration
111    * @return InputFormat
112    */
113   public InputFormat makeInputFormat(Configuration conf) {
114     return ReflectionUtils.newInstance(inputFormatClass, conf);
115   }
116 
117   public InputSplitData getInputSplitData() {
118     return inputSplitData;
119   }
120 
121   public String getLocation() {
122     return location;
123   }
124 
125   @Override
126   public void write(DataOutput out) throws IOException {
127     Writables.writeClassName(out, Preconditions.checkNotNull(inputFormatClass));
128     WritableUtils.writeString(out, Preconditions.checkNotNull(location));
129     inputSplitData.write(out);
130   }
131 
132   @Override
133   public void readFields(DataInput in) throws IOException {
134     inputFormatClass = Writables.readClass(in);
135     location = WritableUtils.readString(in);
136     inputSplitData.readFields(in);
137   }
138 }
