19  package com.facebook.giraph.hive.input.benchmark;
20  
21  import org.apache.hadoop.io.Writable;
22  import org.apache.hadoop.io.WritableComparable;
23  
24  import com.facebook.giraph.hive.input.HiveApiInputObserver;
25  import com.facebook.giraph.hive.record.HiveReadableRecord;
26  import com.yammer.metrics.Metrics;
27  import com.yammer.metrics.core.MetricName;
28  import com.yammer.metrics.core.Timer;
29  import com.yammer.metrics.core.TimerContext;
30  
31  import java.util.concurrent.TimeUnit;
32  
33  /**
34   * Observer that tracks metrics
35   */
36  class MetricsObserver extends HiveApiInputObserver.Empty {
37    private final int printPeriod;
38    /** Timer to read rows */
39    private final Timer readTimer;
40    /** Row reading timer context */
41    private TimerContext readTimerContext;
42    /** Rows read success ratio */
43    private final CounterRatioGauge readSuccessRatio;
44    /** Parse timer */
45    private final Timer parseTimer;
46    /** Parse timer context */
47    private TimerContext parseTimerContext;
48  
49    /**
50     * Constructor
51     *
52     * @param metricsRegistry MetricsRegistry
53     */
54    public MetricsObserver(String name, int printPeriod) {
55      TimeUnit durationUnit = TimeUnit.MICROSECONDS;
56      TimeUnit rateUnit = TimeUnit.MILLISECONDS;
57  
58      this.printPeriod = printPeriod;
59  
60      readTimer = Metrics.newTimer(new MetricName(name, "", "reads"),
61          durationUnit, rateUnit);
62      readSuccessRatio =
63          new CounterRatioGauge(Metrics.newCounter(new MetricName(name, "", "successes")),
64              Metrics.newCounter(new MetricName(name, "", "-reads")));
65      parseTimer = Metrics.newTimer(new MetricName(name, "", "parses"),
66          durationUnit, rateUnit);
67    }
68  
69    public Timer getParseTimer() {
70      return parseTimer;
71    }
72  
73    public CounterRatioGauge getReadSuccessRatio() {
74      return readSuccessRatio;
75    }
76  
77    public Timer getReadTimer() {
78      return readTimer;
79    }
80  
81    @Override
82    public void beginReadRow() {
83      readTimerContext = readTimer.time();
84    }
85  
86    @Override
87    public void endReadRow(WritableComparable key, Writable value) {
88      readSuccessRatio.getNumeratorCounter().inc();
89      readSuccessRatio.getDenominatorCounter().inc();
90      readTimerContext.stop();
91      print(readTimer.count(), printPeriod, "read");
92    }
93  
94    @Override
95    public void hiveReadRowFailed() {
96      readSuccessRatio.getDenominatorCounter().inc();
97      readTimerContext.stop();
98    }
99  
100   @Override
101   public void beginParse() {
102     parseTimerContext = parseTimer.time();
103   }
104 
105   @Override
106   public void endParse(HiveReadableRecord record) {
107     parseTimerContext.stop();
108   }
109 
110   /**
111    * Print information
112    *
113    * @param num Number of rows so far
114    * @param printEvery How often to print
115    * @param prefix String prefix
116    */
117   private static void print(long num, int printEvery, String prefix) {
118     if (num % printEvery == 0) {
119       System.err.println(prefix + " " + num + " records");
120     }
121   }
122 }
