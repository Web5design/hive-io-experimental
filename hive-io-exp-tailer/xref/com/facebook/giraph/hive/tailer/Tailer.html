55    public static void main(String[] args) throws Exception {
56      Opts opts = new Opts();
57      try {
58        Args.parse(opts, args);
59      } catch (IllegalArgumentException e) {
60        LOG.error(e.getMessage());
61        Args.usage(opts);
62        return;
63      }
64      if (opts.help) {
65        Args.usage(opts);
66        return;
67      }
68      opts.process();
69  
70      HadoopNative.requireHadoopNative();
71  
72      HostPort metastoreHostPort = getHostPort(opts);
73      if (metastoreHostPort == null) {
74        return;
75      }
76  
77      ThriftHiveMetastore.Iface client = HiveMetastores.create(metastoreHostPort.host,
78          metastoreHostPort.port);
79  
80      HiveInputDescription inputDesc = new HiveInputDescription();
81      inputDesc.setDbName(opts.database);
82      inputDesc.setTableName(opts.table);
83      inputDesc.setPartitionFilter(opts.partitionFilter);
84      if (opts.requestNumSplits == 0) {
85        opts.requestNumSplits = opts.threads * opts.requestSplitsPerThread;
86      }
87      inputDesc.setNumSplits(opts.requestNumSplits);
88  
89      HiveStats hiveStats = HiveUtils.statsOf(client, inputDesc);
90      LOG.info(hiveStats);
91  
92      HiveConf hiveConf = new HiveConf(Tailer.class);
93      HiveApiInputFormat.setProfileInputDesc(hiveConf, inputDesc, DEFAULT_PROFILE_ID);
94  
95      HiveApiInputFormat hapi = new HiveApiInputFormat();
96      hapi.setMyProfileId(DEFAULT_PROFILE_ID);
97  
98      List<InputSplit> splits = hapi.getSplits(hiveConf, client);
99      LOG.info("Have " + splits.size() + " splits to read");
100 
101     HiveTableName hiveTableName = new HiveTableName(opts.database, opts.table);
102     HiveTableSchema schema = HiveTableSchemas.lookup(client, hiveTableName);
103 
104     Stats stats = new Stats();
105     Context context = new Context(hapi, hiveConf, schema, hiveStats, opts, stats);
106     long startNanos = System.nanoTime();
107 
108     if (opts.threads == 1) {
109       context.splitsQueue = Queues.newArrayDeque(splits);
110       readSplits(context);
111     } else {
112       context.splitsQueue = Queues.newConcurrentLinkedQueue(splits);
113       multiThreaded(context, opts.threads);
114     }
115 
116     long timeNanos = System.nanoTime() - startNanos;
117     stats.printEnd(hiveStats, timeNanos);
118   }

120   private static void multiThreaded(final Context context, int numThreads)
121   {
122     List<Thread> threads = Lists.newArrayList();
123     for (int i = 0; i < numThreads; ++i) {
124       Thread thread = new Thread(new Runnable() {
125         @Override public void run() {
126           readSplits(context);
127         }
128       });
129       thread.setName("readSplit-" + i);
130       thread.start();
131 
132       threads.add(thread);
133     }
134 
135     for (int i = 0; i < threads.size(); ++i) {
136       Uninterruptibles.joinUninterruptibly(threads.get(i));
137     }
138   }

140   private static void readSplits(Context context)
141   {
142     while (context.hasMoreSplitsToRead()) {
143       InputSplit split = context.splitsQueue.poll();
144       try {
145         readSplit(split, context);
146       } catch (Exception e) {
147         LOG.error("Failed to read split " + split, e);
148       }
149     }
150   }

152   private static void readSplit(InputSplit split, Context context)
153       throws IOException, InterruptedException
154   {
155     TaskAttemptID taskId = new TaskAttemptID();
156     TaskAttemptContext taskContext = new TaskAttemptContext(context.hiveConf, taskId);
157     RecordReader<WritableComparable, HiveReadableRecord> recordReader;
158     recordReader = context.hiveApiInputFormat.createRecordReader(split, taskContext);
159     recordReader.initialize(split, taskContext);
160     String[] values = new String[context.schema.numColumns()];
161 
162     int rowsParsed = 0;
163     while (recordReader.nextKeyValue() && !context.limitReached()) {
164       HiveReadableRecord record = recordReader.getCurrentValue();
165       printRecord(record, context.schema.numColumns(), context, values);
166       ++rowsParsed;
167       if (context.rowsParsed.incrementAndGet() > context.opts.limit) {
168         break;
169       }
170       if (rowsParsed % context.opts.metricsUpdatePeriodRows == 0) {
171         context.stats.addRows(context.hiveStats, context.opts.metricsUpdatePeriodRows);
172         rowsParsed = 0;
173       }
174     }
175     context.stats.addRows(context.hiveStats, rowsParsed);
176   }

178   private static void printRecord(HiveReadableRecord record, int numColumns,
179       Context context, String[] values)
180   {
181     StringBuilder sb = context.threadContext.get().stringBuilder;
182     sb.setLength(0);
183     for (int index = 0; index < numColumns; ++index) {
184       if (index > 0) {
185         sb.append(context.opts.separator);
186       }
187       sb.append(String.valueOf(record.get(index)));
188     }
189     System.out.println(sb.toString());
190   }

192   private static HostPort getHostPort(Opts opts) throws IOException {
193     HostPort metastoreHostPort = new HostPort();
194     metastoreHostPort.port = opts.metastorePort;
195     if (opts.metastoreHost != null) {
196       metastoreHostPort.host = opts.metastoreHost;
197     }
198     if (metastoreHostPort.host == null && opts.cluster != null) {
199       if (opts.clustersFile == null) {
200         LOG.error("Cluster file not given");
201         Args.usage(opts);
202         return null;
203       }
204       ObjectMapper objectMapper = new ObjectMapper();
205       File file = new File(opts.clustersFile);
206       ClusterData clustersData = objectMapper.readValue(file, ClusterData.class);
207       List<HostPort> hostAndPorts = clustersData.data.get(opts.cluster);
208       if (hostAndPorts == null) {
209         LOG.error("Cluster " + opts.cluster +
210             " not found in data file " + opts.clustersFile);
211         return null;
212       }
213       Collections.shuffle(hostAndPorts);
214       metastoreHostPort = hostAndPorts.get(0);
215     }
216     return metastoreHostPort;
217   }
